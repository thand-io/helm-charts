{{ if .Values.rbac.create -}}
The Thand Agent is now running with RBAC permissions to manage Kubernetes access.
{{- end }}

1. Check the status of your deployment:
   kubectl get pods --namespace {{ include "agent.namespace" . }} -l "app.kubernetes.io/name={{ include "agent.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

2. View the logs:
   kubectl logs --namespace {{ include "agent.namespace" . }} -l "app.kubernetes.io/name={{ include "agent.name" . }}" -f

3. Access the Thand Agent:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "agent.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "agent.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "agent.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Access the agent at: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "agent.namespace" . }} {{ include "agent.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Access the agent at: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "agent.namespace" . }} -l "app.kubernetes.io/name={{ include "agent.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ include "agent.namespace" . }} port-forward $POD_NAME 8080:{{ .Values.config.server.port }}
  
  Then access the agent at: http://127.0.0.1:8080
{{- end }}

4. Health check endpoints:
   - Health: http://localhost:8080/health
   - Ready: http://localhost:8080/ready
